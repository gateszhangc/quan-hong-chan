### 3.2 埋点追踪逻辑

```typescript
// 点击 Pricing 按钮时的追踪逻辑
function trackPricingView(url?: string, sourceOverride?: string) {
  // 1. 判断是否为 pricing 相关链接
  if (!isPricingUrl(url)) return;
  
  // 2. 确定来源
  const source = sourceOverride || resolvePricingSource();
  //    - header: 从导航栏点击
  //    - faq: 从 FAQ 区域点击
  //    - feature: 从 Features 区域点击
  
  // 3. 发送追踪事件
  trackEvent("pricing_viewed", {
    source: "header",  // 本任务中固定为 header
    timestamp: Date.now(),
    url: url
  });
}
```

## 4. 数据结构

### 4.1 Header 配置结构

```typescript
interface Header {
  brand?: Brand;
  nav?: Nav;
  buttons?: Button[];  // ← 新增
  show_sign?: boolean;
  show_locale?: boolean;
  show_theme?: boolean;
}

interface Button {
  title?: string;      // 按钮文字: "Blog" | "Pricing"
  icon?: string;       // 图标名: "RiArticleLine" | "RiMoneyDollarCircleLine"
  url?: string;        // 链接: "/posts" | "/pricing"
  variant?: "default" | "outline" | "secondary" | "ghost";
  target?: string;     // 打开方式: "_self"
}
```
